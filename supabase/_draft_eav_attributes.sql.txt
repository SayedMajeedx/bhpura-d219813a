-- 1. Dynamic Attribute Definitions Table
CREATE TABLE IF NOT EXISTS public.product_attribute_definitions (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  brand_id uuid NOT NULL REFERENCES public.brands(id) ON DELETE CASCADE,
  code text NOT NULL, -- 'size', 'color', 'fabric', 'flavor', etc.
  name_en text NOT NULL,
  name_ar text,
  type text NOT NULL DEFAULT 'text',
  created_at timestamptz NOT NULL DEFAULT now(),
  updated_at timestamptz NOT NULL DEFAULT now(),
  UNIQUE(brand_id, code)
);

-- Table Grants
GRANT SELECT ON public.product_attribute_definitions TO anon, authenticated;
GRANT ALL ON public.product_attribute_definitions TO service_role, authenticated;

-- Enable RLS
ALTER TABLE public.product_attribute_definitions ENABLE ROW LEVEL SECURITY;

-- Split Policies: Public Select vs Brand Manage
CREATE POLICY "public read active definitions" ON public.product_attribute_definitions
  FOR SELECT TO anon, authenticated
  USING (brand_id IN (SELECT b.id FROM public.brands b WHERE b.is_active = true));

CREATE POLICY "brand manage definitions" ON public.product_attribute_definitions
  FOR ALL TO authenticated
  USING (public.can_access_brand(brand_id))
  WITH CHECK (public.can_access_brand(brand_id));


-- 2. Dynamic Variant Attribute Values Table
CREATE TABLE IF NOT EXISTS public.product_variant_attributes (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  variant_id uuid NOT NULL REFERENCES public.product_variants(id) ON DELETE CASCADE,
  attribute_definition_id uuid NOT NULL REFERENCES public.product_attribute_definitions(id) ON DELETE CASCADE,
  value_en text NOT NULL,
  value_ar text,
  created_at timestamptz NOT NULL DEFAULT now(),
  updated_at timestamptz NOT NULL DEFAULT now(),
  UNIQUE(variant_id, attribute_definition_id)
);

-- Table Grants
GRANT SELECT ON public.product_variant_attributes TO anon, authenticated;
GRANT ALL ON public.product_variant_attributes TO service_role, authenticated;

-- Enable RLS
ALTER TABLE public.product_variant_attributes ENABLE ROW LEVEL SECURITY;

-- Split Policies: Public Select (requires active product & active brand) vs Brand Manage
CREATE POLICY "public read active variant attributes" ON public.product_variant_attributes
  FOR SELECT TO anon, authenticated
  USING (
    EXISTS (
      SELECT 1 FROM public.product_variants pv
      JOIN public.products p ON p.id = pv.product_id
      WHERE pv.id = product_variant_attributes.variant_id 
        AND p.is_active = true -- Guard: draft/hidden products keep attributes hidden
        AND pv.brand_id IN (SELECT b.id FROM public.brands b WHERE b.is_active = true)
    )
  );

CREATE POLICY "brand manage variant attributes" ON public.product_variant_attributes
  FOR ALL TO authenticated
  USING (
    EXISTS (
      SELECT 1 FROM public.product_variants pv
      WHERE pv.id = product_variant_attributes.variant_id AND public.can_access_brand(pv.brand_id)
    )
  )
  WITH CHECK (
    EXISTS (
      SELECT 1 FROM public.product_variants pv
      WHERE pv.id = product_variant_attributes.variant_id AND public.can_access_brand(pv.brand_id)
    )
  );
